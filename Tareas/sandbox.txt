mano(_,0,[]).

mano(Mazo,TamañoMano,Mano) :-
    TamañoMano > 0,
    carta_al_azar(Mazo,MazoRestante,Carta),
    append([Carta],RestoMano,Mano),
    TamañoRestoMano is TamañoMano - 1,
    mano(MazoRestante,TamañoRestoMano,RestoMano).

pred(Mano,Mazo,MazoRestante) :-
    select(Mano,Mazo,MazoRestante).
